  \hline

  3 & Thiết kế Cơ sở dữ liệu (PostgreSQL) &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Thiết kế Schema các bảng lõi: General Ledger (Sổ cái), Voucher (Chứng từ)
    \item Thiết kế các bảng nghiệp vụ: Invoices (Hóa đơn), AR/AP (Công nợ), Receipts/Payments (Thu/Chi)
  \end{itemize} &
  Dương Thanh Toàn &
  File SQL Script khởi tạo Database \& ERD Diagram \\
  \hline

  4 & Phát triển Backend (Core Kế toán) &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Xây dựng API ghi nhận giao dịch (Transaction Recording)
    \item Viết logic tính toán công nợ tự động
    \item Xử lý logic sinh chứng từ và ghi sổ cái (Posting Engine)
  \end{itemize} &
  Dương Thanh Toàn &
  Source code Backend (Controllers, Services, Repositories) \\
  \hline

  5 & Phát triển Frontend (Giao diện) &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Xây dựng giao diện nhập liệu chứng từ (Voucher Entry UI)
    \item Giao diện danh sách \& theo dõi công nợ (AR/AP Tracking UI)
    \item Giao diện xem báo cáo động
  \end{itemize} &
  Cả hai thành viên &
  Source code Frontend (ReactJS Components) \\
  \hline

  6 & Xây dựng Báo cáo Tài chính &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Lập trình xuất Bảng cân đối kế toán, Báo cáo KQKD từ dữ liệu Sổ cái
    \item Tính năng kết xuất ra file Excel/PDF đúng mẫu quy định
  \end{itemize} &
  Dương Thanh Toàn &
  Module Báo cáo (Reporting Service) \\
  \hline

  7 & Tích hợp BI Dashboard &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Kết nối Metabase/Power BI với PostgreSQL
    \item Xây dựng các biểu đồ: Doanh thu, Chi phí, Dòng tiền, Aging Report (Tuổi nợ)
    \item Nhúng (Embed) Dashboard vào ứng dụng Frontend
  \end{itemize} &
  Vũ Đức Toàn &
  Dashboard quản trị trực quan \\
  \hline

  8 & Triển khai AI Chatbot (RAG) &
  \begin{itemize}[leftmargin=*,nosep,topsep=0pt]
    \item Indexing Pipeline: Xây dựng luồng vector hóa dữ liệu hóa đơn, công nợ, báo cáo
    \item Vector DB: Tích hợp Pinecone/FAISS để lưu trữ Embeddings
    \item Chatbot UI: Xây dựng giao diện chat, xử lý luồng hỏi đáp (Truy vấn $\rightarrow$ Truy xuất $\rightarrow$ Sinh câu trả lời)
  \end{itemize} &
  Dương Thanh Toàn &
  Module AI Chatbot hoạt động \& Video demo \\
