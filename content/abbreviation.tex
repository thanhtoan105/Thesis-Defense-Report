% ============ DANH SÁCH TỪ VIẾT TẮT ============

% Sử dụng \newabbreviation thay vì \newacronym

\newabbreviation[
  name={RAG: Retrieval-Augmented Generation},
  description={Kỹ thuật kết hợp truy xuất thông tin từ nguồn bên ngoài với mô hình sinh văn bản để tạo ra câu trả lời chính xác và cập nhật hơn}
]{rag}{RAG}{Retrieval-Augmented Generation}

\newabbreviation[
  name={ERP: Enterprise Resource Planning},
  description={Bộ giải pháp phần mềm tích hợp giúp quản lý đồng bộ các quy trình cốt lõi như tài chính, chuỗi cung ứng và vận hành trong toàn doanh nghiệp}
]{erp}{ERP}{Enterprise Resource Planning}

\newabbreviation[
  name={LLMs: Large Language Models},
  description={Mô hình học sâu được huấn luyện trên tập dữ liệu văn bản khổng lồ để hiểu và sinh ngôn ngữ tự nhiên một cách linh hoạt}
]{llms}{LLMs}{Large Language Models}

\newabbreviation[
  name={RBAC: Role-Based Access Control},
  description={Mô hình kiểm soát truy cập dựa trên vai trò, đảm bảo người dùng chỉ được cấp quyền phù hợp với vai trò và trách nhiệm của họ trong hệ thống}
]{rbac}{RBAC}{Role-Based Access Control}

\newabbreviation[
  name={AP: Accounts Payable},
  description={Quy trình quản lý các khoản nợ phải thanh toán cho nhà cung cấp (công nợ phải trả), gồm kiểm soát hóa đơn, theo dõi hạn mức và thực hiện chi trả}
]{ap}{AP}{Accounts Payable}

\newabbreviation[
  name={GL: General Ledger},
  description={Sổ sách trung tâm ghi nhận toàn bộ giao dịch tài chính của doanh nghiệp, phân loại theo tài khoản để lập báo cáo}
]{gl}{GL}{General Ledger}

\newabbreviation[
  name={CFO: Chief Financial Officer},
  description={Giám đốc tài chính phụ trách hoạch định, kiểm soát dòng tiền và báo cáo tài chính; CRO viết tắt của Chief Risk Officer, người lãnh đạo quản trị rủi ro trong doanh nghiệp}
]{cfo}{CFO}{Chief Financial Officer}

\newabbreviation[
  name={UI: Giao diện người dùng},
    description={Các thành phần trực quan và tương tác của sản phẩm mà người dùng thao tác (ví dụ: điều khiển, bố cục)}
]{ui}{UI}{User Interface}

\newabbreviation[
  name={BI: Business Intelligence},
  description={Hệ thống thu thập, xử lý và phân tích dữ liệu nhằm hỗ trợ doanh nghiệp ra quyết định nhanh chóng, chính xác và dựa trên bằng chứng}
]{bi}{BI}{Business Intelligence}

\newabbreviation[
  name={CSAT: Customer Satisfaction Score},
  description={Chỉ số đo lường mức độ hài lòng của khách hàng đối với sản phẩm hoặc dịch vụ thông qua khảo sát đánh giá}
]{csat}{CSAT}{Customer Satisfaction Score}

\newabbreviation[
  name={SAP: Systems Applications and Products in Data Processing},
  description={Giải pháp phần mềm quản trị doanh nghiệp tích hợp giúp quản lý tài chính, chuỗi cung ứng, sản xuất, nguồn nhân lực và quan hệ khách hàng trên một nền tảng thống nhất}
]{sap}{SAP}{Systems Applications and Products in Data Processing}

\newabbreviation[
  name={SQL: Ngôn ngữ truy vấn có cấu trúc},
  description={Ngôn ngữ dùng để quản lý và truy vấn dữ liệu trong cơ sở dữ liệu quan hệ (ví dụ: SELECT, INSERT, UPDATE, DELETE)}
]{sql}{SQL}{Structured Query Language}

\newabbreviation[
  name={API: Giao diện lập trình ứng dụng},
  description={Tập hợp các quy tắc và công cụ giúp các thành phần phần mềm giao tiếp và tích hợp với nhau}
]{api}{API}{Application Programming Interface}

\newabbreviation[
  name={ACID: Atomicity, Consistency, Isolation, Durability},
  description={Tập thuộc tính đảm bảo giao dịch cơ sở dữ liệu thực thi đáng tin cậy: toàn vẹn, nhất quán, cô lập và bền vững}
]{acid}{ACID}{Atomicity, Consistency, Isolation, Durability}
